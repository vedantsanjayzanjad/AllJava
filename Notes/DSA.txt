*************  Singly Linked List
Singly Linked List is used to stored collections of node and has follwoing properties

DATA | NEXT =>LIST NODE (NEXT == NULL) for last node
1)It Contains Sequence of node.
2)A node has data and refrence to the next list node.
3)First node is the head node.
4)Last node has the data and have points to null.

10|  5|  8|null

(IMPLEMENTATION)
package test;

import java.util.Arrays;

package test;

public class SingleLinkedList {

	private ListNode head;

	private static class ListNode {
		private int data;
		private ListNode next;

		public ListNode(int data) {
			this.data = data;
			this.next = null;
		}
	}

	public void display() {
		ListNode currentNode=head;
		while(currentNode != null) {
			System.out.print(currentNode.data + "--->");
			currentNode = currentNode.next ;
		}
		System.out.println("null");
	}

	public static void main(String[] args) {
		SingleLinkedList singleLinkedList = new SingleLinkedList();
		singleLinkedList.head = new ListNode(10);
		ListNode second = new ListNode(5);
		ListNode third = new ListNode(9);
		ListNode fourth = new ListNode(7);

		// Now will connect together them to form a chain
		singleLinkedList.head.next = second;
		second.next = third;
		third.next = fourth;
		singleLinkedList.display();

		// 10--->5----->9----->7------>null
	}
}
